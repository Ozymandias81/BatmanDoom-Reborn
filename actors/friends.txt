ACTOR MayorHostage replaces Candelabra
{
	//$Category Reborn NPCs
	Tag "Major Kroll (Hostage)"
	Health 200
	Radius 24
	Height 56
	Mass 1000000
	Speed 0
	PainChance 200
	+FLOORCLIP
	+SHOOTABLE
	+SOLID
	SpawnID 1
	SeeSound "cyber/death"
	PainSound "knight/pain"
	DeathSound "weapons/rocklx"
	States
		{
		Spawn:
			HEAD F 10 A_Look
			Loop
		See:
			HEAD F 50
			HEAD E 12
			HEAD F 12 A_PlaySound("vile/start")
			Goto See+1
		Pain:
			MANF B 7 A_Pain
			Goto See+1
		Death:
			VILE R 5 Bright
			VILE S 5 Bright A_Scream
			VILE TUVWX 5 Bright
			VILE YZ 5
			VILE Z -1 A_BrainDie
			Stop
		}
}
